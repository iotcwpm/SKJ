77  void get(uint time, const Model& model){
78  uint year = IOSKJ::year(time);
79  uint quarter = IOSKJ::quarter(time);
80 
81  // Maldive PL quarterly CPUE
82  if(year>=2000 and year<=2013){
83  // Just get M/PL vulnerable biomass
84  m_pl_cpue(year,quarter) = model.biomass_vulnerable(M,PL) * model.m_pl_quarter(quarter);
85 
86  // At end, scale expected by geometric mean over period 2004-2012
87  if(year==2013 and quarter==3){
88  GeometricMean geomean;
89  for(uint year=2004;year<=2012;year++){
90  for(uint quarter=0;quarter<4;quarter++){
91  geomean.append(m_pl_cpue(year,quarter));
92  }
93  }
94  double scaler = 1/geomean.result();
95  for(auto& fit : m_pl_cpue) fit *= scaler;
96  }
97  }
98 
99  // West PS annual CPUE
100  if(year>=1985 and year<=2013){
101  // Currently take a mean of vulnerable biomass over all quarters in the year...
102  static Array<double,Quarter> cpue_quarters;
103  // ... get this quarter's CPUE and save it
104  cpue_quarters(quarter) = model.biomass_vulnerable(W,PS);
105  // ... if this is the last quarter then take the geometric mean
106  if(quarter==3){
107  w_ps_cpue(year) = geomean(cpue_quarters);
108  }
109 
110  // At end, scale expected by geometric mean over period 1991-2010
111  if(year==2013 and quarter==3){
112  GeometricMean geomean;
113  for(uint year=1991;year<=2010;year++){
114  geomean.append(w_ps_cpue(year,quarter));
115  }
116  double scaler = 1/geomean.result();
117  for(auto& fit : w_ps_cpue) fit *= scaler;
118  }
119  }
120 
121  // Size frequencies by region and method
122  if(year>=1982 and year<=2013){
123  // Generate expected size frequencies for each method in each
124  // region regardless of whether there is observed data or not
125  for(auto region : regions){
126  for(auto method : methods){
127  Array<double,Size> composition = 0;
128  // Calculate selected numbers by size accumulated over ages
129  for(auto size : sizes){
130  for(auto age : ages){
131  composition(size) += model.numbers(region,age,size) * model.selectivities(method,size);
132  }
133  }
134  // Proportionalise
135  composition /= sum(composition);
136  // Store
137  for(auto size : sizes) size_freqs(year,quarter,region,method,size) = composition(size);
138  }
139  }
140  }
141 
142  // Z-estimates for W region
143  if(year>=2005 and year<=2013){
144  // Generate expected values of Z for each size bin
145  // Expected values of Z are calculated by combining natural mortality and
146  // fishing mortality rates
147  for(auto z_size : z_sizes){
148  // Model size classes are 2mm wide, so for each of the 5mm wide Z-estimate bins there are three model
149  // size classes to average over e.g.
150  // 45-50 Z-estimate size bin ~ 45,47,49 model size class mid points ~ ([44,46,48])/2 ~ 22,23,24 size dimension levels
151  // Calculate the mean Z for the model size classes in each Z-estimate size bin
152  double z = 0;
153  uint z_lower = 45+z_size.index()*5;
154  uint size_class = (z_lower-1)/2;
155  for(uint size=size_class;size<size_class+3;size++){
156  double survival = model.mortalities_survival(size) * model.exploitation_survival(W,size);
157  // Capture cases where survuval is estimated to be zero to prevent overflow
158  if(survival>0) z += -log(survival);
159  else z += -log(0.000001);
160  }
161  z /= 3;
162  // Store
163  z_ests(year,quarter,z_size) = z;
164  }
165  }
