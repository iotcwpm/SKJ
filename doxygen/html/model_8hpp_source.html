278  // Initialise arrays by size...
279  for(auto size : sizes){
280 
281  lengths[size] = 2*size+1;
282 
283  weights[size] = weight_a * std::pow(lengths[size],weight_b);
284 
285  maturities[size] = 1/(1+std::pow(19,(maturity_inflection-lengths[size])/maturity_steepness));
286  }
287 
288  // Initialise natural survival rate
289  survival = std::exp(-0.25*mortality);
290 
291  // Initialise growth size transition matrix
292  for(auto size : sizes){
293  growth_increments(size) = (growth_assymptote-lengths(size))*(1-std::exp(-0.25*growth_rate));
294  }
295  for(auto size_from : size_froms){
296  double growth_increment = growth_increments(size_from);
297  double mean = lengths(size_from) + growth_increment;
298  double sd = std::pow(std::pow(growth_sd,2)+std::pow(growth_increment*growth_cv,2),0.5);
299  for(auto size : sizes){
300  growth(size_from,size) = normal_integral(2*size,2*(size+1),mean,sd);
301  }
302  }

355  void step(void){
356 
357  // Recruits
358  if(recruit_relation_on){
359  // Stock-recruitment realtion is active so calculate recruits based on
360  // the spawning biomass in the previous time step
362  recruits_determ = 4 * recruit_steepness * recruit_virgin * biomass_spawning_overall / (
363  (5*recruit_steepness-1)*biomass_spawning_overall + recruit_virgin_spawners*(1-recruit_steepness)
364  );
365  } else {
366  // Stock-recruitment relation is not active so recruitment is just r0.
367  recruits_determ = recruit_virgin;
368  }
369  if(recruit_variation_on){
370  recruits_deviation = 1;
371  } else {
372  recruits_deviation = 1;
373  }
374  recruits = recruits_determ * recruits_deviation;
375 
376  if(exploitation_on){
377  // Exploitation rate
378  for(auto region : regions){
379  for(auto method : methods){
380  double sum = 0;
381  for(auto age : ages){
382  for(auto size : sizes){
383  sum += numbers(region,age,size) * weights(size) * selectivities(method,size);
384  }
385  }
386  biomass_vulnerable(region,method) = sum;
387  }
388  }
389  }
390 
391  // Ageing and recruitment
392  for(auto region : regions){
393  for(auto size : sizes){
394  // Oldest age class accumulates
395  numbers(region,ages.size-1,size) += numbers(region,ages.size-2,size);
396 
397  // For most ages just "shuffle" along
398  for(uint age=ages.size-2;age>0;age--){
399  numbers(region,age,size) = numbers(region,age-1,size);
400  }
401

403  // according to `initials`
404  numbers(region,0,size) = recruits * recruit_regions(region) * recruit_sizes(size);
405  }
406  }
407 
408  // Mortality, growth and movement
409  auto numbers_temp = numbers;
410  for(auto region : regions){
411  for(auto age : ages){
412  for(auto size : sizes){
413  double sum = 0;
414  for(auto region_from : region_froms){
415  for(auto size_from : size_froms){

417  growth(size_from,size) *

420  movement(region_from,region);
421  }

424  }
