36  std::vector<Y> b(n), d(n), a(n), c(n+1), l(n+1), u(n+1), z(n+1);

39  l[0] = Y(1);
40  u[0] = Y(0);
41  z[0] = Y(0);
42  h[0] = x[1] - x[0];
43 
44  for (size_type i = 1; i < n; i++) {
45  h[i] = x[i+1] - x[i];
46  l[i] = Y(2 * (x[i+1] - x[i-1])) - Y(h[i-1]) * u[i-1];
47  u[i] = Y(h[i]) / l[i];
48  a[i] = (Y(3) / Y(h[i])) * (y[i+1] - y[i]) - (Y(3) / Y(h[i-1])) * (y[i] - y[i-1]);
49  z[i] = (a[i] - Y(h[i-1]) * z[i-1]) / l[i];
50  }
51 
52  l[n] = Y(1);
53  z[n] = c[n] = Y(0);
54 
55  for (size_type j = n-1; j >= 0; j--) {
56  c[j] = z[j] - u[j] * c[j+1];
57  b[j] = (y[j+1] - y[j]) / Y(h[j]) - (Y(h[j]) * (c[j+1] + Y(2) * c[j])) / Y(3);
58  d[j] = (c[j+1] - c[j]) / Y(3 * h[j]);
59  }
