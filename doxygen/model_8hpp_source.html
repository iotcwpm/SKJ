467  void initialise(void){
468  // Initialise grids by size...
469  for(auto size : sizes){
470  double length = 2*size.index()+1;
471  lengths(size) = length;
472  weights(size) = weight_length_a*std::pow(length,weight_length_b);
473  maturities(size) = 1.0/(1.0+std::pow(19,(maturity_length_inflection-length)/maturity_length_steepness));
474  }
475 
476  // Mortality at size
477  // The maximum mortality is determined by the mortality_cap which defines
478  // the _size_ at which mortality is highest. Because of this, mortalities needs
479  // to be initalised after weights have been initialised
480  for(auto size : sizes){
481  double mortality_weight = (size<mortality_cap_size)?weights(mortality_cap_size):weights(size);
482  mortalities(size) = mortality_base * std::pow(mortality_weight,mortality_exponent);
483  mortalities_survival(size) = std::exp(-0.25*mortalities(size));
484  }
485 
486  // Initialise proportion of recruits by size
487  Normal recruits_lengths_dist(
488  recruits_lengths_mean,
489  recruits_lengths_mean*recruits_lengths_cv
490  );
491  for(auto size : sizes){
492  double length = lengths(size);
493  recruits_sizes(size) = recruits_lengths_dist.integral(length-1,length+1);
494  };
495 
496  // Initialise growth size transition matrix
497  for(auto size : sizes){
498  double increment = (growth_assymptote-lengths(size))*(1-std::exp(-0.25*growth_rate));
499  growth_increments(size) = (increment>0)?increment:0;
500  }
501  for(auto size_from : size_froms){
502  Level<Size> size_from_size(size_from);
503  double length_from = lengths(size_from_size);
504  double growth_increment = growth_increments(size_from_size);
505  double mean = length_from + growth_increment;
506  double sd = std::pow(std::pow(growth_sd,2)+std::pow(growth_increment*growth_cv,2),0.5);
507  Normal distribution(mean,sd);
508  for(auto size : sizes){
509  double length_to = lengths(size);
510  growth(size_from,size) = distribution.integral(length_to-1,length_to+1);
511  }
512  }
513 
514  // Initialise movement matrix
515  // Normalise so that movement proportions always sum to 1 for a particular region
516  auto movement_sums = movement_pars(sum,by(region_froms));
517  for(auto region_from : region_froms){
518  for(auto region : regions){
519  movement(region_from,region) =
520  movement_pars(region_from,region)/movement_sums(region_from);
521  }
522  }
523 
524  // Initialise selectivity
525  for(auto method : methods){
526  // Iterpolate with piecewise spline
527  // Ensure that selectivity is between 0 and 1 since spline
528  // can produce values outside this range even if knots are not
529  double max = 0;
530  for(auto size : sizes){
531  double length = lengths(size);
532  double selectivity;
533  if(length<selectivity_lengths(0)) selectivity = 0;
534  else{
535  for(uint knot=0;knot<selectivity_knots.size()-1;knot++){
536  if(selectivity_lengths(knot)<=length and length<selectivity_lengths(knot+1)){
537  selectivity = selectivity_values(method,knot) + (length-selectivity_lengths(knot)) * (
538  selectivity_values(method,knot+1)-selectivity_values(method,knot))/(
539  selectivity_lengths(knot+1)-selectivity_lengths(knot)
540  );
541  }
542  }
543  }
544  if(selectivity<0) selectivity = 0;
545  else if(selectivity>max) max = selectivity;
546  selectivities(method,size) = selectivity;
547  }
548  for(auto size : sizes) selectivities(method,size) /= max;
549  }
550 
551  // Normalise the recruits_region grid so that it sums to one
552  recruits_regions /= recruits_regions(sum);
553 
554  // Initialise recruits_variation
555  recruits_variation = Lognormal(1,recruits_sd);
556 
557  // During debug mode dump the model here for easy inspection
558  // Done here before equilibrium() in case that fails
559  #if DEBUG
560  write();
561  #endif
562 
563  // Go to pristine
564  pristine_go();
565 
566  // During debug mode dump the model here for easy inspection
567  // Done after equilibrium() and biomass_spawning_unfished has been set
568  #if DEBUG
569  write();
570  #endif
571  }

576  void update(uint time){
577  uint year = IOSKJ::year(time);
578  uint quarter = IOSKJ::quarter(time);
579 
580  // Calculate total biomass, spawners biomass and spawning biomass by region
581  for(auto region : regions){
582  double biomass_ = 0;
583  double biomass_spawners_ = 0;
584  double biomass_spawning_ = 0;
585  for(auto age : ages){
586  for(auto size : sizes){
587  double biomass = numbers(region,age,size) * weights(size)/1000;
588  biomass_ += biomass;
589  double spawners = biomass * maturities(size);
590  biomass_spawners_ += spawners;
591  biomass_spawning_ += spawners * spawning(quarter);
592  }
593  }
594  biomass(region) = biomass_;
595  biomass_spawners(region) = biomass_spawners_;
596  biomass_spawning(region) = biomass_spawning_;
597 
598  }
599  biomass_spawning_overall(quarter) = biomass_spawning(sum);
600 
601  // Recruits
602  if(recruits_relation_on){
603  // Stock-recruitment relation is active so calculate recruits based on
604  // the spawning biomass in the previous time step
605  double h = recruits_steepness;
606  double r0 = recruits_unfished;
607  double s0 = biomass_spawning_unfished(quarter);
608  double s = biomass_spawning_overall(quarter);
609  recruits_determ = 4*h*r0*s/((5*h-1)*s+s0*(1-h));
610  } else {
611  // Stock-recruitment relation is not active so recruitment is just r0.
612  recruits_determ = recruits_unfished;
613  }
614 
615  if(recruits_variation_on){
616  recruits_deviation = recruits_variation.random();
617  }
618 
619  recruits = recruits_determ * recruits_deviation;
620 
621  // Ageing and recruitment
622  for(auto region : regions){
623  for(auto size : sizes){
624  // Oldest age class accumulates
625  numbers(region,ages.size()-1,size) += numbers(region,ages.size()-2,size);
626 
627  // For most ages just "shuffle" along
628  for(uint age=ages.size()-2;age>0;age--){
629  numbers(region,age,size) = numbers(region,age-1,size);
630  }
631 
632  // Recruits are evenly distributed over regions and over sizes
633  // according to `initials`
634  numbers(region,0,size) = recruits * recruits_regions(region) * recruits_sizes(size);
635  }
636  }
637 
638  if(exploit!=exploit_none){
639  // For each region and method
640  for(auto region : regions){
641  for(auto method : methods){
642  // Calculate vulnerable biomass
643  double biomass_vuln = 0;
644  for(auto age : ages){
645  for(auto size : sizes){
646  biomass_vuln += numbers(region,age,size) *
647  weights(size)/1000 *
648  selectivities(method,size);
649  }
650  }
651  biomass_vulnerable(region,method) = biomass_vuln;
652 
653  // Update CPUE
654  if(quarter==0){
655  if(year==1985) cpue_base(region,method).reset();
656  if(year>=1985 and year<=2013){
657  cpue_base(region,method).append(biomass_vuln);
658  } else {
659  cpue(region,method) = biomass_vuln/cpue_base(region,method);
660  }
661  }
662 
663  // Determine exploitation rate
664  double er = 0;
665  if(exploit==exploit_catch){
666  // Calculate exploitation rate from catches and biomass_vulnerable
667  double c = catches(region,method);
668  if(c>0){
669  if(biomass_vuln>0){
670  er = c/biomass_vuln;
671  if(er>1) er = 1;
672  } else er = 1;
673  } else er = 0;
674  // Update estimate of catchability
675  double e = effort(region,method);
676  if(e>0){
677  double q = er/e;
678  if(year==2004) catchability_estim(region,method).reset();
679  if(q>0 and year>=2004 and year<=2013) catchability_estim(region,method).append(q);
680  if(year==2013){
681  catchability(region,method) = catchability_estim(region,method);
682  // Where no catches for a region method make catchability 0
683  if(not std::isfinite(catchability(region,method))){
684  catchability(region,method) = 0;
685  }
686  }
687  }
688  }
689  else if(exploit==exploit_effort){
690  // Calculate exploitation rate from number of
691  // effort units
692  er = catchability(region,method) * effort(region,method);
693  } else {
694  // Use the exploitation rate specified
695  er = exploitation_rate_specified(region,method);
696  }
697  exploitation_rate(region,method) = er;
698  catches_taken(region,method) = er * biomass_vuln;
699  }
700  }
701  // Pre-calculate the exploitation_survival for each region and size
702  for(auto region : regions){
703  for(auto size : sizes){
704  double proportion_taken = 0;
705  for(auto method : methods){
706  proportion_taken += exploitation_rate(region,method) * selectivities(method,size);
707  }
708  exploitation_survival(region,size) = (proportion_taken>1)?0:(1-proportion_taken);
709  }
710  }
711  } else {
712  exploitation_survival = 1;
713  }
714 
715  // Mortality, growth and movement
716  auto numbers_temp = numbers;
717  for(auto region : regions){
718  for(auto age : ages){
719  for(auto size : sizes){
720  double number = 0;
721  for(auto region_from : region_froms){
722  for(auto size_from : size_froms){
723  Level<Region> rf(region_from);
724  Level<Size> sf(size_from);
725  number += numbers(rf,age,sf) *
726  growth(size_from,size) *
727  mortalities_survival(sf) *
728  exploitation_survival(rf,sf) *
729  movement(region_from,region);
730  }
731  }
732  numbers_temp(region,age,size) = number;
733  }
734  }
735  }
736  numbers = numbers_temp;
737  }

743  void equilibrium(void){
744  #if DEBUG
745  std::cout<<"************equilibrium()**************\n";
746  #endif
747  // Turn off recruitment variation (saving current setting
748  // for putting back at the end of this method)
749  bool recruits_variation_on_setting = recruits_variation_on;
750  recruits_variation_on = false;
751  // Seed the population with a small population in each partition
752  numbers = 1;
753  // Iterate until there is a very minor change in biomass
754  uint steps = 0;
755  const uint steps_max = 1000;
756  Array<double,Region> biomass_prev = 1;
757  while(steps<steps_max){
758  // It is necessary to update the model for each quarter so that quarterly differences
759  // in dynamics (e.g. spawning proportion) are incorporated
760  for(uint quarter=0;quarter<4;quarter++) update(quarter);
761 
762  // Break if biomass has gone to very low levels (as happens when this method
763  // is called with high exploitation rates from yield_curve) since the proportional
764  // diffs minimise to a low level
765  if(biomass(sum)<0.01) break;
766 
767  double diffs = 0;
768  for(auto region : regions){
769  diffs += fabs(biomass(region)-biomass_prev(region))/biomass_prev(region);
770  }
771  diffs /= regions.size();
772  if(diffs<0.0001) break;
773  biomass_prev = biomass;
774 
775  #if DEBUG
776  std::cout<<steps<<"\t"<<biomass(W)<<"\t"<<biomass(M)<<"\t"<<biomass(E)<<"\t"<<diffs<<std::endl;
777  #endif
778 
779  // Throw an error if undefined biomass
780  if(not std::isfinite(biomass(W)+biomass(M)+biomass(E))) throw std::runtime_error("Biomass is not finite. Check inputs");
781 
782  steps++;
783  }
784  // Throw an error if there was no convergence
785  if(steps>steps_max) throw std::runtime_error("No convergence in equilibrium() function");
786  // Turn on recruitment deviation again
787  recruits_variation_on = recruits_variation_on_setting;
788  }
