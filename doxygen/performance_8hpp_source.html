166  void record(uint time, const Model& model){
167  times.append();
168 
169  // Catch magnitude
170  auto catch_total = model.catches_taken(sum);
171  catches_total.append(catch_total);
172  auto catches_by_method = model.catches_taken(sum,by(methods));
173  catches_ps.append(catches_by_method(PS));
174  catches_pl.append(catches_by_method(PL));
175  catches_gn.append(catches_by_method(GN));
176 
177  // Catch variability
178  if(catch_total>0){
179  catches_var.append(catch_total);
180  catches_mapc.append(catch_total);
181  }
182  catches_shut.append(catch_total==0);
183 
184  // Stock status relative to unfished
185  auto status = model.biomass_status(time);
186  status_mean.append(status);
187  status_b10.append(status<0.1);
188  status_b20.append(status<0.2);
189 
190  // Biomass relative to Bmsy
191  auto b = model.biomass_spawners(sum)/model.biomass_spawners_msy;
192  b_ratio.append(b);
193  // F relative to Fmsy
194  auto f = model.fishing_mortality_get()/model.f_msy;
195  f_ratio.append(f);
196 
197  // Kobe plot
198  // Determine quadrant
199  char quadrant;
200  if(b>=1){
201  if(f<=1) quadrant = 'a';
202  else quadrant = 'b';
203  } else {
204  if(f<=1) quadrant = 'c';
205  else quadrant = 'd';
206  }
207  // Update performance measures for proportion of time spent in each quadrant
208  kobe_a.append(quadrant=='a');
209  kobe_b.append(quadrant=='b');
210  kobe_c.append(quadrant=='c');
211  kobe_d.append(quadrant=='d');
212  // Update performance measure for time taken to get back into quadrant A
213  if(quadrant=='a'){
214  // If previously outside of A then append the time that
215  // have been outside to the mean and reset time counter to zero.
216  if(kobe_out_a>0){
217  kobe_to_a.append(kobe_out_a);
218  kobe_out_a = 0;
219  }
220  } else {
221  // Outside of A so increment time counter.
222  kobe_out_a++;
223  }
224 
225  // Catch rates
226  // Use vulnerable (i.e. selected) biomass for the three main regions/gears
227  // relative to the start year as a measure of catch rates (CPUE)
228  if(times==1){
229  // Record baselines
230  for(auto region : regions){
231  for(auto method : methods){
232  cpue_baseline(region,method) = model.biomass_vulnerable(region,method);
233  }
234  }
235  } else {
236  // Record relative to baseline
237  for(auto region : regions){
238  for(auto method : methods){
239  cpue_mean(region,method).append(
240  model.biomass_vulnerable(region,method)/cpue_baseline(region,method)

242  }
