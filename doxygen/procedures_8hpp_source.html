378  int quarter = IOSKJ::quarter(time);
379  // Operate once per year in the third quarter
380  if(quarter==0){
381  // Get CPUE as a combination of W/PS and M/PL
382  GeometricMean combined;
383  combined.append(model.cpue(W,PS));
384  combined.append(model.cpue(M,PL));
385  double cpue = combined;
386  // Add observation error
387  Lognormal imprecision(1,precision);
388  cpue *= imprecision.random();
389  // Update smoothed index
390  if(index_==-1) index_ = cpue;
391  else index_ = responsiveness*cpue + (1-responsiveness)*index_;
392  // Calculate recommended harvest rate
393  double rate;
394  if(index_<limit) rate = 0;
395  else if(index_>threshold) rate = multiplier;
396  else rate = multiplier/(threshold-limit)*(index_-limit);
397  // Calculate recommended TAC
398  double tac = std::min(rate*cpue,maximum);
399  // Restrict changes in TAC
400  if(last_!=-1){
401  double change = tac/last_;
402  double max = 1 + change_max;
403  if(change>max) change = max;
404  else if(change<1/max) change = 1/max;
405  tac = last_*change;
406  }
