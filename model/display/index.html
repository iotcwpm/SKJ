plot_track <- function(what,label){
	print(
		ggplot(track,aes(x=year+quarter/4)) + 
			geom_line(aes_string(y=what)) + 
			geom_hline(y=0,alpha=0) +
			geom_vline(x=2015,alpha=0.6,linetype=2) +
			labs(x='Year',y=label)
	)
}
